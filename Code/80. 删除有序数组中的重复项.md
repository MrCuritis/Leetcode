### 代码
``` cpp
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int sameCnt = 0;
        int curNum = -1;
        for(int i = 0;i<nums.size();++i){
            if(curNum!=nums[i]){
                curNum = nums[i];
                sameCnt = 0;
                continue;
            }
            ++sameCnt;
            if(sameCnt>=2){
                nums.erase(nums.begin()+i);
                --i;
            }
        }
        return nums.size();

    }
};
```
