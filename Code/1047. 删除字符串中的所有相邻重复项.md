### 题解
用栈即可。

### 代码
``` cpp
class Solution {
public:
    string removeDuplicates(string s) {
        std::stack<char> sk;
        std::string res;
        int i = 0;
        while(i < s.size()){
            if(sk.empty() || sk.top() != s[i]){
                sk.push(s[i]);
            }else{
                sk.pop();
            }
            ++i;
        }

        while(!sk.empty()){
            res += sk.top();
            sk.pop();
        }
        std::reverse(res.begin(), res.end());
        return res;
    }
};
```
