class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> record = new HashMap<Integer, Integer>();
        int num = 0;
        int[] result =new int[2];
        for(int i = 0; i < nums.length; i++){
            num = target - nums[i];
            if(record.containsKey(num)){
                result[0] = record.get(num);
                result[1] = i;
                return result;
            }
            else{
                record.put(nums[i], i);
            }
        }
        return result;
    }
}
